app.delete('/book/:isbn',[
//Set form validation rule
check('isbn')
.isLength({ min: 5 })
.isNumeric()
.custom(value => {
return book.findOne({where: {isbn: value}}).then(b => {
if(!b){
throw new Error('ISBN not found');
}
})
}
),
], (req, res) => {
book.destroy({where: {isbn: req.params.isbn}})
.then(affectedRow => {
if(affectedRow){
return {
"status":"success",
"message": "Book deleted",
"data": null
}
}
return {
"status":"error",
"message": "Failed",
"data": null
}
})
.then(r => {
res.json(r)
})
})
